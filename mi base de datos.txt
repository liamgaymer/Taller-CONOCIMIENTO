CREATE DATABASE IF NOT EXISTS fashiondb;
USE fashiondb;

-- Crear tablas
CREATE TABLE IF NOT EXISTS Colecciones (
    codigo_coleccion VARCHAR(10) PRIMARY KEY,
    nombre VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS Prendas (
    codigo_prenda VARCHAR(10) PRIMARY KEY,
    nombre VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS Coleccion_Prendas (
    codigo_coleccion VARCHAR(10),
    codigo_prenda VARCHAR(10),
    PRIMARY KEY (codigo_coleccion, codigo_prenda),
    FOREIGN KEY (codigo_coleccion) REFERENCES Colecciones(codigo_coleccion),
    FOREIGN KEY (codigo_prenda) REFERENCES Prendas(codigo_prenda)
);

-- Creamos una tabla que ya tenga el Tipo y la Temporada
DROP TABLE IF EXISTS ResumenVentas;

CREATE TABLE ResumenVentas (
    prenda_tipo VARCHAR(50),
    temporada VARCHAR(50),
    monto_venta DECIMAL(10, 2)
);

-- Metemos datos de ejemplo 
INSERT INTO ResumenVentas VALUES 
('Camisa', 'Verano', 50.00),
('Camisa', 'Verano', 30.00), -- Otra camisa de verano
('Pantalón', 'Verano', 100.00),
('Camisa', 'Invierno', 80.00);

-- LA CONSULTA QUE HACE EL TRABAJO
SELECT prenda_tipo, temporada, SUM(monto_venta)
FROM ResumenVentas
GROUP BY prenda_tipo, temporada;
